#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,i,l;
    string s;
    cin>>t;
    for(i=1 ; i<=t ; i++)
    {
        cin>>s;
        l=s.size();
        if(l>10)
        {
            cout<<s[0]<<l-2<<s[l-1]<<endl;
        }
        else
        {
            cout<<s<<endl;
        }
    }
    return 0;
}
