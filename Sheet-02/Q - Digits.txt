#include<bits/stdc++.h>
using namespace std;
int main()
{
   int test,i,x,rem;
   cin>>test;
   for(i=1 ; i<=test ; i++)
   {
       cin>>x;
       if(x==0)cout<<x;
       while(x!=0)
       {
           rem=x%10;
           x=x/10;
           cout<<rem<<" ";
       }
       cout<<endl;
   }
    return 0;
}
