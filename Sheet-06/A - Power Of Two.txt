#include<stdio.h>
#include<math.h>
int main()
{
    long long int n,i=0,a,j=0;
    scanf("%lld",&n);
    while(1)
    {
        a=pow(2,i);
        i++;
        if(a==n)
        {
            j=1;
            break;
        }
        else if(a>n)
        {
            j=0;
            break;
        }
    }
    if(j==1)printf("YES");
    else printf("NO");
}
